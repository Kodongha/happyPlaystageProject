DROP TABLE USER_INFO cascade constraints;

    USER_EMAIL    VARCHAR2(50) NOT NULL,
    USER_PWD    VARCHAR2(100) NOT NULL,
    USER_NM    VARCHAR2(30) NOT NULL,

COMMENT ON COLUMN USER_INFO.USER_PHONE IS '휴대전화';

COMMENT ON COLUMN USER_INFO.USER_GRADE_CD IS '회원등급코드';

COMMENT ON COLUMN USER_INFO.ENROLL_DT IS '회원가입일자';

COMMENT ON COLUMN USER_INFO.SNS_CD IS 'SNS코드';

COMMENT ON COLUMN USER_INFO.LEAVE_TF IS '탈퇴여부';

COMMENT ON COLUMN USER_INFO.LEAVE_DT IS '탈퇴일자';
